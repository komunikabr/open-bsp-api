const res = await fetch('https://api.msghub.app/v1/messages', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    to: '+5511999999999',
    channel: 'whatsapp',
    text: 'Olá! Como posso ajudar?'
  })
})
const data = await res.json()
import requests

res = requests.post('https://api.msghub.app/v1/messages',
    headers={'Authorization': 'Bearer YOUR_API_KEY'},
    json={
        'to': '+5511999999999',
        'channel': 'whatsapp',
        'text': 'Olá! Como posso ajudar?'
    }
)
data = res.json()
curl -X POST https://api.msghub.app/v1/messages \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{"to":"+5511999999999","channel":"whatsapp","text":"Olá!"}'
